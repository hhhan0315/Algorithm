```js
let num = 5;
// 4 1
// 3 3
// 2 5
// 1 7
// 0 9 -> 이렇게 출력

function makeStar(num) {
  for (let n = 0; n < num; n++) {
    let result = "";
    for (let i = 0; i < num - n; i++) {
      result += " ";
    }
    for (let j = 0; j < 2 * n + 1; j++) {
      result += "*";
    }
    console.log(result);
  }
}

makeStar(num);
```
