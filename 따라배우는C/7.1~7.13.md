## 7.2 표준 입출력 함수들 getchar(), putchar() 예제
```java
#include <stdio.h>	// getchar(), putchar()

int main() {
	/* 1. getchar(), putchar() : 한 글자씩 처리*/
	char ch;
	ch = getchar();	// 한 글자 받아와서 char에 넣기	if) 입력 abcde -> 출력 a
	putchar(ch);	// 출력


	/* 2. while문으로 문장 받음*/
	char ch2;

	while ((ch2 = getchar()) != '\n') {	// 입력 abcde -> 출력 abcde
		putchar(ch2);	// buffer로 인해 한번에 받아서 출력해줌
	}


	/* 3.특정문자 Filter*/
	char ch3;

	while ((ch3 = getchar()) != '\n') {
		if (ch3 == 'f' || ch3 == 'F')	// f 입력하지 않기
			ch3 = 'X';
		putchar(ch3);
	}


	/* 4.숫자 -> 별표 교체*/
	char ch4;

	while ((ch4 = getchar()) != '\n') {
		if ('0' <= ch4 && ch4 <= '9')
			ch4 = '*';
		putchar(ch4);
	}

	/* 5.소문자 -> 대문자*/
	char ch5;

	while ((ch5 = getchar()) != '\n') {
		if ('a' <= ch5 && ch5 <= 'z')
			ch5 += ('A'-'a');
		putchar(ch5);
	}

	return 0;
}
```

## 7.2 ctype.h 문자 함수들
```java
#include <stdio.h>
#include <ctype.h>	// isalpha() and more

int main() {
	char ch;

	while ((ch = getchar()) != '\n') {
		/* 대문자 소문자 변경*/
		if (islower(ch))
			ch = toupper(ch);
		else if (isupper(ch))
			ch = tolower(ch);

		/* 숫자 -> 별표*/
		if (isdigit(ch) != 0)	// == (isdigit(ch))
			ch = '*';

		putchar(ch);
	}

	return 0;
}
```
## 7.6 소수 판단 예제
```c
#include <stdio.h>
#include <stdbool.h>

int main() {
	unsigned num, div;

	scanf("%u", &num);

	bool isPrime = true; // flag 의미 (상태 임시 표현), int isPrime=0; 으로도 사용
	for (div = 2; (div * div) <= num; div++) {	// (div*div) 사용으로 반복 줄일 수 있다.
		if (num % div == 0) {
			isPrime = false;

			if (num == div * div) {
				printf("%u is divisible by %u.\n", num, div);
			}
			else {
				printf("%u is divisible by %u and %u.\n", num, div, num / div);
			}
		}
	}

	if (isPrime)
		printf("%u is a prime number.\n", num);
	else
		printf("%u is not a prime number.\n", num);

	return 0;
}
```

## 7.8 단어 세기 예제
```c
#include <stdio.h>
#include <ctype.h>
#include <stdbool.h>
#define STOP '.'

int main() {
	char ch;
	int c_char = 0, c_word = 0, c_line = 0;
	bool line_flag = false;	// flag 활용을 잘하자.
	bool word_flag = false;

	printf("Enter text : \n");

	while ((ch = getchar()) != STOP) {
		if (!isspace(ch)) {	// 나는 isalpha(ch)
			c_char++;
		}

		if (!isspace(ch) && !word_flag) {
			c_word++;
			word_flag = true;
		}

		if (isspace(ch)) {
			word_flag = false;
		}

		if (!isspace(ch) && !line_flag) {
			c_line++;
			line_flag = true;
		}

		if (ch == '\n') {
			line_flag = false;
		}
	}

	printf("Characters = %d, Words = %d, Lines = %d\n", c_char, c_word, c_line);
	return 0;
}

```
