### 76. 이항계수(메모이제이션)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int memo[21][21];

int dfs(int n, int r)
{
	if (memo[n][r] > 0) return memo[n][r];
	if (n == r || r == 0) return 1;
	else
	{
		return memo[n][r] = dfs(n - 1, r - 1) + dfs(n - 1, r);
	}
	// 5C3 = 4C2(내가 들어감) + 4C3(내가 들어가지 않음)
}

int main()
{
	int N, R;
	cin >> N >> R;

	cout << dfs(N, R) << endl;

	return 0;
}
```
