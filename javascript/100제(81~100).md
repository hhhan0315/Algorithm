function math(e) {
  let stack = [];
  const check = { ")": "(", "}": "{" };

  for (let i in e) {
    if (e[i] === "(" || e[i] === "{") {
      stack.push(e[i]);
    } else if (check[e[i]]) {
      if (stack.length === 0) {
        return false;
      } else {
        let t = check[e[i]];
        if (t !== stack.pop()) {
          return false;
        }
      }
    }
  }

  return stack.length === 0;
}

let input = "{()()()}";
console.log(math(input));

const n = 2;

function solution(chars) {
  let permute = [];

  const f = (prefix, chars) => {
    for (let i = 0; i < chars.length; i++) {
      permute.push(prefix + chars[i]);

      f(prefix + chars[i], chars.slice(i + 1));
    }
  };
  f("", chars);
  const result = permute.filter((x) => x.length === n);
  result.sort((a, b) => b - a);

  return result[0];
}

function solution(input, dish) {
  dish -= 1;
  let count = 0;

  let s = input.slice();

    console.log(input);
    let p = input.shift();
    if (s[0] === p) {
      if (dish === 0) break;

      s.shift();
    } else {
      input.push(p);

        dish -= 1;
      }
    }
    count++;
  }

}
