### 21. 카드게임
```c
#include <iostream>

using namespace std;

int main()
{
	int A[10];
	int B[10];

	for (int i = 0; i < 10; i++)
	{
		cin >> A[i];
	}

	for (int i = 0; i < 10; i++)
	{
		cin >> B[i];
	}

	int count_a = 0, count_b = 0;
	int last_win = 0;

	for (int i = 0; i < 10; i++)
	{
		if (A[i] > B[i])
		{
			count_a+=3;
			last_win = 1;
		}
		else if (A[i] < B[i])
		{
			count_b+=3;
			last_win = 2;
		}
		else
		{
			count_a += 1;
			count_b += 1;
		}
	}

	cout << count_a << " " << count_b << endl;

	if (count_a == count_b)
	{
		if (last_win == 1)
			cout << "A" << endl;
		else if (last_win == 2)
			cout << "B" << endl;
		else
			cout << "D" << endl;
	}
	else if (count_a > count_b)
	{
		cout << "A" << endl;
	}
	else
	{
		cout << "B" << endl;
	}

	return 0;
}
```

### 22. 온도의 최댓값
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N, K, sum = 0, max;
	cin >> N >> K;
	vector<int> arr(N);         // 동적 할당

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	for (int i = 0; i < K; i++)
	{
		sum += arr[i];      // 첫번째 구간 합
	}
	max = sum;

	for (int i = K; i < N; i++)
	{
		sum -= arr[i - K];  // 생각하지 못했다.
		sum += arr[i];
		if (sum > max)
			max = sum;
	}
	cout << max << endl;

	//*틀린이유* : O(n^2)은 시간 초과, O(n)으로 해결.
	return 0;
}
```

### 23. 연속 부분 증가수열
```c
#include <iostream>
using namespace std;

int main()
{
	int N, pre, now, cnt, max;
	cin >> N;

	cin >> pre;
	cnt = 1;
	max = 1;

	for (int i = 1; i < N; i++)
	{
		cin >> now;
		if (now >= pre)
		{
			cnt++;
			if (cnt > max)
				max = cnt;
		}
		else
		{
			cnt = 1;
		}
		pre = now;

	}
	cout << max << endl;

	return 0;
}
```

### 24. Jolly Jumpers
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int N, pre, now;
	bool flag = true;
	cin >> N;
	vector<int> v(N);

	cin >> pre;
	for (int i = 1; i < N; i++)
	{
		cin >> now;
		int num = abs(pre - now);
		if (num > 0 && num < N && v[num] == 0)	// *틀린이유* : 범위와 문제 조건
		{
			v[num] = 1;
		}
		else
		{
			cout << "NO" << endl;
			return 0;
		}
		pre = now;
	}
	cout << "YES" << endl;

	return 0;
}
```

### 25. 석차 구하기
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N;
	cin >> N;
	vector<int> v(N);
	vector<int> rank(N);

	for (int i = 0; i < N; i++)
	{
		cin >> v[i];
		rank[i] = 1;
	}

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			if (v[i] < v[j])
				rank[i]++;	// *틀린이유* : 생각하지 못함.
		}
	}

	for (int i = 0; i < N; i++)
	{
		cout << rank[i] << " ";
	}

	return 0;
}
```

### 26. 마라톤
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N, cnt, now;
	cin >> N;
	vector<int> v(N);

	for (int i = 0; i < N; i++)
	{
		cin >> v[i];
	}

	for (int i = 0; i < N; i++)
	{
		cnt = 1;
		now = v[i];
		for (int j = i - 1; j >= 0; j--)
		{
			if (v[j] >= now)
			{
				cnt++;
			}
		}
		cout << cnt << " ";
	}

	return 0;
}
```

### 27. N!의 표현법
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N, temp, div;
	cin >> N;
	vector<int> vec(N+1);

	for (int i = 2; i <= N; i++)
	{
		temp = i;	// *틀린이유* temp, div 초기값 설정
		div = 2;
		while (true)
		{
			if (temp % div == 0)
			{
				temp /= div;
				vec[div]++;
			}
			else
			{
				div++;
			}
			if (temp == 1) break;
		}
	}
	cout << N << "! = ";
	for (int i = 2; i <= N; i++)
	{
		if (vec[i] != 0)
			cout << vec[i] << " ";
	}

	return 0;
}
```

### 28. N!에서 0의 개수
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N, cnt1 = 0, cnt2 = 0, temp, div;
	cin >> N;

	for (int i = 2; i <= N; i++)
	{
		temp = i;
		div = 2;
		while (true)
		{
			if (temp % div == 0)
			{
				if (div == 2) cnt1++;
				else if (div == 5) cnt2++;
				temp /= div;

			}
			else
			{
				div++;
			}
			if (temp == 1) break;
		}
	}

	if (cnt1 < cnt2)
		cout << cnt1 << endl;
	else
		cout << cnt2 << endl;

	return 0;
}
```
