### 83. 복면산 (send+more=money)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int a[8];	//arr[8] , D E M N O R S Y
int visited[10];

int send()
{
	return a[6] * 1000 + a[1] * 100 + a[3] * 10 + a[0];
	// S, E, N, D 배열 순서
}
int more()
{
	return a[2] * 1000 + a[4] * 100 + a[5] * 10 + a[1];
}
int money()
{
	return a[2] * 10000 + a[4] * 1000 + a[3] * 100 + a[1] * 10 + a[7];
}

void dfs(int L)
{
	if (L == 8)
	{
		if (send() + more() == money())
		{
			if (a[2] == 0 || a[6] == 0) return;	// 첫 자리가 0이면 return.
			cout <<"  "<< a[6] << " " << a[1] << " " << a[3] << " " << a[0] << endl;
			cout <<"+ "<< a[2] << " " << a[4] << " " << a[5] << " " << a[1] << endl;
			cout <<"-----------" << endl;
			cout << a[2] << " " << a[4] << " " << a[3] << " " << a[1] <<" " <<a[7]<<endl;
		}
	}
	else
	{
		for (int i = 0; i < 10; i++)
		{
			if (visited[i] == 0)
			{
				a[L] = i;
				visited[i] = 1;
				dfs(L+1);
				visited[i] = 0;
			}
		}
	}
}
// 순열로 해당되는 수 구함.

int main()
{
	dfs(0);

	return 0;
}
```

### 84. 휴가 (삼성 DFS)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> T, P;
int res = 0, N;

void dfs(int L, int sum)
{
	if (L == N + 1)
	{
		if (sum > res)
			res = sum;
	}
	else
	{
		// if 조건을 생각하지 못했다.
		if (L + T[L] <= N + 1) dfs(L + T[L], sum + P[L]);
		dfs(L + 1, sum);
	}
}

int main()
{
	int a, b;
	cin >> N;

	T.push_back(0);	// 0번 index : 0
	P.push_back(0);

	for (int i = 0; i < N; i++)
	{
		cin >> a >> b;
		T.push_back(a);	// 1번 index부터 사용.
		P.push_back(b);
	}

	dfs(1, 0);
	cout << res << endl;

	return 0;
}
```

### 85. 수식만들기 (삼성 DFS)
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
using namespace std;
vector<int> v, op;
int N;
int MAX = INT_MIN;
int MIN = INT_MAX;

void dfs(int L, int sum)
{
	if (L == N)
	{
		if (sum < MIN)
			MIN = sum;
		if (sum > MAX)
			MAX = sum;
	}
	else
	{
		if (op[0] > 0)
		{
			op[0]--;
			dfs(L + 1, sum + v[L]);
			op[0]++;
		}
		if (op[1] > 0)
		{
			op[1]--;
			dfs(L + 1, sum - v[L]);
			op[1]++;
		}
		if (op[2] > 0)
		{
			op[2]--;
			dfs(L + 1, sum * v[L]);
			op[2]++;
		}
		if (op[3] > 0)
		{
			op[3]--;
			dfs(L + 1, sum / v[L]);
			op[3]++;
		}
	}
}

int main()
{
	int num;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> num;
		v.push_back(num);
	}

	for (int i = 0; i < 4; i++)	// 연산자 개수
	{
		cin >> num;
		op.push_back(num);
	}

	dfs(1, v[0]);
	cout << MAX << endl;
	cout << MIN << endl;

	return 0;
}
```

### 86. 피자 배달 거리 (삼성 DFS)
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
using namespace std;
vector<pair<int, int> > home, pizza;
int ch[12], M, res = INT_MAX;

void dfs(int s, int L)
{
	if (L == M)
	{
		int sum = 0;
		for (int i = 0; i < home.size(); i++)
		{
			int x1 = home[i].first;
			int y1 = home[i].second;
			int dist = INT_MAX;
			for (int j = 0; j < M; j++)
			{
				int x2 = pizza[ch[j]].first;
				int y2 = pizza[ch[j]].second;
				dist = min(dist, abs(x1 - x2) + abs(y1 - y2));
			}
			sum += dist;
		}
		if (sum < res)
			res = sum;
	}
	else
	{
		for (int i = s; i < pizza.size(); i++)	// 조합 구하기
		{
			ch[L] = i;
			dfs(i + 1, L + 1);
		}
		// 0 1 2 3 -> 0 1 2 4 -> 0 1 2 5 이렇게 ch[] 배열에 저장하며 동작한다.
		// 후에 각 피자의 좌표 계산을 위해 활용.
	}
}

int main()
{
	int N, temp;
	cin >> N >> M;

	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			cin >> temp;
			if (temp == 1) home.push_back(make_pair(i, j));	// 좌표 입력
			else if (temp == 2) pizza.push_back(make_pair(i, j));
		}
	}

	dfs(0, 0);
	cout << res << endl;

	return 0;
}
```

### 87. 섬나라 아일랜드 (BFS)
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;
int map[21][21];
int dx[8] = { 1,1,0,-1,-1,-1,0,1 };
int dy[8] = { 0,-1,-1,-1,0,1,1,1 };
int cnt = 0;

struct loc
{
	int x;
	int y;
	loc(int a, int b)
	{
		x = a;
		y = b;
	}
};

int main()
{
	int N;
	cin >> N;

	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			cin >> map[i][j];
		}
	}

	queue<loc> Q;

	// 저번에 풀어봤는데 이중 for문을 해주지 않고 dx[4], dy[4]로 지정해줘서 틀림
	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			if (map[i][j] == 1)
			{
				Q.push(loc(i, j));
				map[i][j] = 0;

				while (!Q.empty())
				{
					loc temp = Q.front();
					Q.pop();
					for (int i = 0; i < 8; i++)
					{
						int nx = temp.x + dx[i];
						int ny = temp.y + dy[i];
						if (map[nx][ny] == 1)
						{
							Q.push(loc(nx, ny));
							map[nx][ny] = 0;
						}
					}
				}
				cnt++;
			}
		}
	}

	cout << cnt << endl;

	return 0;
}
```

### 88. 미로의 최단거리 통로 (BFS)
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;

struct loc
{
	int x;
	int y;
	loc(int a, int b)
	{
		x = a;
		y = b;
	}
};

int main()
{
	int size = 7;
	int map[21][21];
	int dis[21][21];

	int dx[4] = { -1,1,0,0 };
	int dy[4] = { 0,0,-1,1 };

	for (int i = 1; i <= size; i++)
	{
		for (int j = 1; j <= size; j++)
		{
			cin >> map[i][j];
		}
	}

	for (int i = 1; i <= size; i++)
	{
		for (int j = 1; j <= size; j++)
		{
			dis[i][j] = 0;
		}
	}

	queue<loc> Q;
	Q.push(loc(1, 1));
	map[1][1] = 1;

	while (!Q.empty())
	{
		loc temp = Q.front();
		Q.pop();
		for (int i = 0; i < 4; i++)
		{
			int nx = temp.x + dx[i];
			int ny = temp.y + dy[i];
			if (map[nx][ny] == 0 && nx >= 1 && nx <= size && ny >= 1 && ny <= size)
			{
				Q.push(loc(nx, ny));
				map[nx][ny] = 1;
				dis[nx][ny] = dis[temp.x][temp.y] + 1;
			}
		}
	}

	cout << dis[size][size] << endl;

	return 0;
}
```
