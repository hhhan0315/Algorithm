## 8.2 캡슐화, 접근 지정자, 접근 함수
```c
#include <iostream>
#include <string>
#include <vector>
using namespace std;

class Date
{
// 기본은 private (멤버 변수는 private으로 사용하는 것이 좋다.)
	int m_month;
	int m_day;
	int m_year;

public:
	// access function
	void setDate(const int& month_input, const int& day_input, const int& year_input)
	{
		m_month = month_input;
		m_day = day_input;
		m_year = year_input;
	}

	void setMonth(const int& month_input)
	{
		m_month = month_input;
	}

	const int& getMonth()	// int getMonth()에서 수정.
	{
		return m_month;
	}

	void copyFrom(const Date& original)
	{
		m_month = original.m_month;
		m_day = original.m_day;
		m_year = original.m_year;
	}
};

int main()
{
	Date today;// { 2, 12, 2020 };	// private이라 못함.
	// 그냥 값을 수정하는 것은 오류 발생 가능성이 커진다.
	/*today.m_month = 2;*/
	today.setDate(2, 12, 2020);
	today.setMonth(8);

	cout << today.getMonth() << endl;

	Date copy;
	copy.copyFrom(today);

	return 0;
}
```

## 8.3 생성자 Constructors
```c
#include <iostream>
using namespace std;

class Fraction
{
private:
	int m_numerator;
	int m_denominator;

public:
	// 생성자 Constructors
	Fraction()
	{
		m_numerator = 1;
		m_denominator = 1;
		cout << "Fraction() constructor" << endl;
	}

	Fraction(const int& num_in, const int& den_in)
	{
		m_numerator = num_in;
		m_denominator = den_in;
		cout << "Fraction() constructor" << endl;
	}

	void print()
	{
		cout << m_numerator << " / " << m_denominator << endl;
	}
};

int main()
{
	Fraction frac;	// 외워두기 // frac : 생성자의 parameter가 없을 경우 괄호를 제거.(함수와 비교가 힘들어서)
	frac.print();

	Fraction fracTwo(1, 3);
	fracTwo.print();

	Fraction fracThree{ 1.0,3 };	// 엄격하게 double 1.0 잡아서 failed
	Fraction fracThree(1.0, 3);	// warning 호출하고 실행은 함.

	return 0;
}
```

## 8.5 위임 생성자
```c
#include <iostream>
using namespace std;

class Student
{
private:
	int	m_id;
	string	m_name;

public:
	Student(const string& name_input)
		//:Student(0, name_input)	// 위임생성자
	{
		init(0, name_input);
	}

	Student(const int& id_input, const string& name_input)
		//:m_id(id_input), m_name(name_input)
	{
		init(id_input, name_input);

	{
		m_id = id_input;
		m_name = name_input;
	}

	void print()
	{
		cout << m_id << " " << m_name << endl;
	}
};

int main()
{
	Student student1{ 1,"seung" };
	student1.print();

	Student student2{ "kevin" };
	student2.print();

	return 0;
}
```

## 8.6 소멸자 destructor
```c
#include <iostream>
using namespace std;

class IntArray
{
private:
	int* m_arr = nullptr;
	int m_length = 0;

public:
	IntArray(const int length_in)
	{
		m_length = length_in;
		m_arr = new int[m_length];

		cout << "Constructor " << endl;
	}
	// 동적할당과 연결해서 소멸자가 중요한 역할은 한다.
	~IntArray()
	{
		if (m_arr != nullptr)
			delete[] m_arr;
	}

	int getLength() { return m_length; }
};

int main()
{
	while (true)
	{
		IntArray my_int_arr(10000);
	}

	return 0;
}
```

## 8.9 클래스와 const
```c
#include <iostream>
using namespace std;

class Something
{
public:
	int m_value = 0;

	// 기본적으로 숨겨져있다.
	Something(const Something& st_in)
	{
		m_value = st_in.m_value;
		cout << "Copy constructor" << endl;
	}

	Something()
	{
		cout << "Constructor" << endl;
	}

	void setValue(int value)	// setValue()는 const 할 수 없다.
	{
		m_value = value;
	}
	int getValue() const
	{
		return m_value;
	}
};

void print(const Something &st)	// 매개변수를 reference로 사용.
{
	cout << &st << endl;
	cout << st.getValue() << endl;
}

int main()
{
	const Something something;
	//something.setValue(3);
	//cout << something.getValue() << endl;

	// 주소가 같다.
	cout << &something << endl;

	print(something);

	return 0;
}
```
```c
#include <iostream>
using namespace std;

class Something
{
public:
	string m_value = "default";

	const string& getValue() const
	{
		cout << "const" << endl;
		return m_value;
	}
	string& getValue()
	{
		cout << "non-const" << endl;
		return m_value;
	}
};

int main()
{
	Something something;	// non-const
	something.getValue() = "hi";
	cout << something.getValue() << endl;

	const Something something2;	// const
	something2.getValue();

	// 오버로딩 시 매개변수가 다르면 오버로딩이 가능하다고 했다.
	// const와 non-const 차이로 오버로딩이 가능한지 아닌지도 알고 있으면 좋다.

	return 0;
}
```

## 8.10 정적 멤버 변수
```c
#include <iostream>
using namespace std;

class Something
{
public:
	static int s_value;
	// static const int s_value=1; 이면 main()에서 const라서 값 수정 불가.
};

int Something::s_value = 1;	// cpp에 static 정의해야함.

int main()
{
	Something s1;
	Something s2;

	s1.s_value = 2;
	cout << &s1.s_value << " " << s1.s_value << endl;
	cout << &s2.s_value << " " << s2.s_value << endl;

	return 0;
}
```
