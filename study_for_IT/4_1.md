### 76. 이항계수(메모이제이션)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int memo[21][21];

int dfs(int n, int r)
{
	if (memo[n][r] > 0) return memo[n][r];
	if (n == r || r == 0) return 1;
	else
	{
		return memo[n][r] = dfs(n - 1, r - 1) + dfs(n - 1, r);
	}
	// 5C3 = 4C2(내가 들어감) + 4C3(내가 들어가지 않음)
}

int main()
{
	int N, R;
	cin >> N >> R;

	cout << dfs(N, R) << endl;

	return 0;
}
```

### 77. 친구인가? (Union & Find)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int unf[1001];

int Find(int v)
{
	if (v == unf[v]) return v;
	else return unf[v] = Find(unf[v]);	// 경로 압축
	//else return Find(unf[v]);
}

void Union(int a, int b)
{
	a = Find(a);
	b = Find(b);
	if (a != b) unf[a] = b;
}

int main()
{
	int N, M, a, b, x, y;
	cin >> N >> M;

	for (int i = 1; i <= N; i++)
	{
		unf[i] = i;
	}

	for (int i = 0; i < M; i++)
	{
		cin >> a >> b;
		Union(a, b);
	}

	cin >> x >> y;
	int res1 = Find(x);
	int res2 = Find(y);
	if (res1 == res2) cout << "YES" << endl;
	else cout << "NO" << endl;

	for (int i = 1; i <= 9; i++)
	{
		cout << unf[i] << " ";
	}

	return 0;
}
```

### 78. 원더랜드 (Kruskal MST), union & find
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int unf[101];

struct Edge {
	int start, end, weight;
	Edge(int a, int b, int c)
	{
		start = a;
		end = b;
		weight = c;
	}
	bool operator<(Edge& e)
	{
		return weight < e.weight;
	}
};

int Find(int v)
{
	if (v == unf[v]) return v;
	else return Find(unf[v]);
}

void Union(int a, int b)
{
	a = Find(a);
	b = Find(b);
	if (a != b) unf[a] = b;
}

int main()
{
	int V, E, A, B, C, res = 0;
	vector<Edge> v;
	cin >> V >> E;

	for (int i = 1; i <= V; i++)
	{
		unf[i] = i;
	}

	for (int i = 0; i < E; i++)
	{
		cin >> A >> B >> C;
		v.push_back(Edge(A, B, C));
	}
	sort(v.begin(), v.end());

	for (int i = 0; i < E; i++)
	{
		int fa = Find(v[i].start);
		int fb = Find(v[i].end);
		if (fa != fb)
		{
			res += v[i].weight;
			Union(v[i].start, v[i].end);
		}
	}
	cout << res << endl;

	return 0;
}
```

### 79. 원더랜드 (Prim MST), priority_queue
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;
int visited[30];

struct Edge
{
	int node;
	int weight;
	Edge(int a, int b)
	{
		node = a;
		weight = b;
	}
	bool operator<(const Edge& b)const
	{
		return weight > b.weight;	// 최소 힙
	}	//반대로는 최대 힙
};

int main()
{
	int V, E, A, B, C, res = 0;
	cin >> V >> E;
	priority_queue<Edge> PQ;
	vector<pair<int, int> > map[30];

	for (int i = 0; i < E; i++)
	{
		cin >> A >> B >> C;
		map[A].push_back(make_pair(B, C));	// 무방향 가중치
		map[B].push_back(make_pair(A, C));
	}

	PQ.push(Edge(1, 0));	// 시작 정점 : 1, 가중치 : 0

	while (!PQ.empty())
	{
		Edge temp = PQ.top();
		PQ.pop();
		int v = temp.node;
		int cost = temp.weight;

		if (visited[v] == 0)
		{
			res += cost;
			visited[v] = 1;
			for (int i = 0; i < map[v].size(); i++)
			{
				PQ.push(Edge(map[v][i].first, map[v][i].second));
			}
		}
	}

	cout << res << endl;

	return 0;
}
/*
(1,0) -> pop(1, 0)
push(2,12), push(9, 25) -> pop(2,12), res=12 -> push(3,10), push(8,17), push(9,25) 반복*/
```

### 80. 다익스트라 알고리즘
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
using namespace std;
int visited[21];
int dist[21];

int main()
{
	int N, M, a, b, c;
	cin >> N >> M;
	vector<pair<int, int> > map[21];

	for (int i = 0; i < M; i++)
	{
		cin >> a >> b >> c;
		map[a].push_back(make_pair(b, c));
	}

	for (int i = 0; i <= N; i++)
	{
		dist[i] = INT_MAX;
	}
	dist[1] = 0;

	for (int i = 1; i <= N; i++)
	{
		int min = 0;
		for (int j = 1; j <= N; j++)
		{
			if (visited[j] == 0 && dist[j] < dist[min])
				min = j;
		}
		visited[min] = 1;

		for (int j = 0; j < map[min].size(); j++)
		{
			int v = map[min][j].first;
			int cost = map[min][j].second;
			if (dist[v] > dist[min] + cost)
			{
				dist[v] = dist[min] + cost;
			}
		}
	}

	for (int i = 2; i <= N; i++)
	{
		if (dist[i] != INT_MAX)
		{
			cout << i << " : " << dist[i] << endl;
		}
		else
			cout << i << " : impossible" << endl;
	}

	return 0;
}
```

### 81. 벨만-포드 알고리즘
```c
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
using namespace std;
int dist[101];

struct Edge
{
	int start, end, weight;
	Edge(int a, int b, int c)
	{
		start = a;
		end = b;
		weight = c;
	}
};

int main()
{
	vector<Edge> Ed;
	int N, M, a, b, c, s, e;
	cin >> N >> M;
	for (int i = 0; i < M; i++)
	{
		cin >> a >> b >> c;
		Ed.push_back(Edge(a, b, c));
	}

	for (int i = 1; i <= N; i++)
	{
		dist[i] = INT_MAX;
	}

	cin >> s >> e;
	dist[s] = 0;
	for (int i = 1; i < N; i++)	// 간선 1개부터~N개일 때 가는 최소값
	{
		for (int j = 0; j < Ed.size(); j++)
		{
			int s = Ed[j].start;
			int e= Ed[j].end;
			int w = Ed[j].weight;
			if (dist[s] != INT_MAX && dist[s] + w < dist[e])
			{
				dist[e] = dist[s] + w;
			}
		}
	}

	for (int j = 0; j < Ed.size(); j++)	// 음의 사이클이 있다.
	{
		int s = Ed[j].start;
		int e = Ed[j].end;
		int w = Ed[j].weight;
		if (dist[s] != INT_MAX && dist[s] + w < dist[e])
		{
			cout << "-1" << endl;
			exit(0);
		}
	}

	cout << dist[e] << endl;

	return 0;
}
```
