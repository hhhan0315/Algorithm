## 9.1 산술 연산자 오버로딩 하기
```c
#include <iostream>
using namespace std;

class Cents
{
private:
	int m_cents;

public:
	Cents(int cents = 0) { m_cents = cents; }
	int getCents() const { return m_cents; }
	int& getCents() { return m_cents; }

	// friend 함수로 오버로딩
	friend Cents operator + (const Cents& c1, const Cents& c2)
	{
		return Cents(c1.getCents() + c2.getCents());
	}

	// 멤버 함수로 구현(=, [], (), ->) 이러한 형태는 멤버 함수로만 오버로딩 가능.
	Cents operator + (const Cents& c2)
	{
		return Cents(this->m_cents + c2.m_cents);
	}
};

// 일반적 함수로 오버로딩
//Cents operator + (const Cents& c1, const Cents& c2)
//{
//	return Cents(c1.getCents() + c2.getCents());
//}

int main()
{
	Cents cents1(6);
	Cents cents2(8);

	// 계산에 위험이 있다면 add처럼 함수 사용하는 것이 좋다.
	// cout << add(cents1, cents2).getCents() << endl;
	cout << (cents1 + cents2 + Cents(6)).getCents() << endl;	// 20

	return 0;
}
```

## 9.3 단항 연산자 오버로딩 하기
```c
#include <iostream>
using namespace std;

class Cents
{
private:
	int m_cents;

public:
	Cents(int cents = 0) { m_cents = cents; }
	int getCents() const { return m_cents; }
	int& getCents() { return m_cents; }

	Cents operator - () const
	{
		return Cents(-m_cents);
	}

	bool operator ! () const
	{
		// 조건은 목적에 따라 정하기
		return (m_cents == 0) ? true : false;
	}

	friend std::ostream& operator <<(std::ostream& out, const Cents& cents)
	{
		out << cents.m_cents;
		return out;
	}
};

int main()
{
	Cents cents1(6);
	Cents cents2(0);

	cout << cents1 << endl;        // 6
	cout << -cents1 << endl;       // -6
	cout << -Cents(-10) << endl;   // 10

	cout << !cents1 << " " << !cents2 << endl;	// 0 1

	return 0;
}
```

## 9.4 비교 연산자 오버로딩 하기
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

class Cents
{
private:
	int m_cents;

public:
	Cents(int cents = 0) { m_cents = cents; }
	int getCents() const { return m_cents; }
	int& getCents() { return m_cents; }

	friend bool operator == (const Cents& c1, const Cents& c2)	// ==, != 가능.
	{
		return c1.m_cents == c2.m_cents;
	}

	friend bool operator < (const Cents& c1, const Cents& c2)	// std::sort에서는 < 를 구현.
	{
		return c1.m_cents < c2.m_cents;
	}

	friend std::ostream& operator <<(std::ostream& out, const Cents& cents)
	{
		out << cents.m_cents;
		return out;
	}
};

int main()
{
	Cents cents1(6);
	Cents cents2(6);

	if (cents1 == cents2)
		cout << "equal" << endl;

	vector<Cents> arr(20);
	for (unsigned i = 0; i < 20; ++i)
		arr[i].getCents() = i;

	std::random_shuffle(begin(arr), end(arr));

	for (auto& e : arr)
		cout << e << " ";
	cout << endl;

	std::sort(begin(arr), end(arr));

	// 비교 operator를 오버로딩 해주어야 실행 가능.
	for (auto& e : arr)
		cout << e << " ";
	cout << endl;

	return 0;
}
```

## 9.5 증감 연산자 오버로딩 하기
```python
#include <iostream>
using namespace std;

class Digit
{
private:
	int m_digit;

public:
	Digit(int digit = 0) : m_digit(digit) {}

	// prefix : parameter가 비어져있다.
	Digit& operator ++ ()
	{
		++m_digit;
		return *this;
	}

	// postfix : parameter에 dummy로 아무거나 들어간다.
	Digit operator ++ (int)
	{
		Digit temp(m_digit);
		++(*this);
		return temp;
	}

	friend std::ostream& operator <<(std::ostream& out, const Digit& d)
	{
		out << d.m_digit;
		return out;
	}
};

int main()
{
	Digit d(5);
	cout << ++d << endl;	// 6
	cout << d++ << endl;	// 6
	cout << d << endl;	// 7

  	return 0;
}
```

## 9.7 괄호 연산자 오버로딩과 함수 객체
```c

{
private:
	int m_counter = 0;
```

## 9.8 형변환을 오버로딩 하기
```c
#include <iostream>
using namespace std;

class Cents
{
private:
	int m_cents;
public:
	Cents(int cents = 0)
	{
		m_cents = cents;
	}

	operator int()
	{
		cout << "cast here" << endl;
		return m_cents;
	}
};

class Dollar
{
private:
	int m_dollars = 0;
public:
	Dollar(const int& input):m_dollars(input){}

	operator Cents()
	{
		return Cents(m_dollars * 100);
	}
};

void printInt(const int& value)
{
	cout << value << endl;
}

int main()
{
	Dollar dollar(2);
	Cents cents = dollar;
	printInt(cents);	// 200

	return 0;
}
```
