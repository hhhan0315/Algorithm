### 32. 선택 정렬
```c
#include <iostream>
using namespace std;

int main()
{
	int arr[100];
	int N;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	int least, temp;
	for (int i = 0; i < N - 1; i++)
	{
		least = i;
		for (int j = i + 1; j < N; j++)
		{
			if (arr[j] < arr[least])
			{
				least = j;
			}
		}
		temp = arr[least];
		arr[least] = arr[i];
		arr[i] = temp;
	}

	for (int i = 0; i < N; i++)
	{
		cout << arr[i] << " ";
	}

	return 0;
}
```

### 33. 3등의 성적은? (정렬 응용)
```c
#include <iostream>
using namespace std;

int main()
{
	int arr[100];
	int N;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	int least, temp;
	for (int i = 0; i < N - 1; i++)
	{
		least = i;
		for (int j = i + 1; j < N; j++)
		{
			if (arr[j] > arr[least])
			{
				least = j;
			}
		}
		temp = arr[least];
		arr[least] = arr[i];
		arr[i] = temp;
	}

	int grade = 1;
	for (int i = 1; i < N; i++)
	{
		if (arr[i-1] != arr[i])
		{
			grade++;
		}
		if (grade == 3)
		{
			cout << arr[i] << endl;
			break;
		}
	}

	return 0;
}
```

### 34. 버블 정렬
```c
#include <iostream>
using namespace std;

int main()
{
	int arr[100];
	int N;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	int temp;
	for (int i = 0; i < N - 1; i++)
	{
		for (int j = 0; j < N - i - 1; j++)
		{
			if (arr[j] > arr[j + 1])
			{
				temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}

	for (int i = 0; i < N; i++)
	{
		cout << arr[i] << " ";
	}

	return 0;
}
```

### 35. Special sort (버블 정렬 응용)
```c
#include <iostream>
using namespace std;

int main()
{
	int arr[100];
	int N;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	int temp;
	for (int i = 0; i < N - 1; i++)
	{
		for (int j = 0; j < N - i - 1; j++)
		{
			if (arr[j] > 0 && arr[j + 1] < 0)	// 앞에가 양수, 뒤에가 음수 일때 자리 교체
			{
				temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}

	for (int i = 0; i < N; i++)
	{
		cout << arr[i] << " ";
	}

	return 0;
}
```

### 36. 삽입 정렬
```c
#include <iostream>
using namespace std;

int main()
{
	int arr[100];
	int N;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	int temp, j;
	for (int i = 1; i < N; i++)
	{
		temp = arr[i];
		for (j = i - 1; j >= 0; j--)
		{
			if (arr[j] > temp)
			{
				arr[j + 1] = arr[j];
			}
			else
				break;
		}
		arr[j + 1] = temp;
	}

	for (int i = 0; i < N; i++)
	{
		cout << arr[i] << " ";
	}

	return 0;
}
```

### 37. Least Recently Used (삽입 정렬 응용)
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int S, N, i, j, k, temp, pos;
	cin >> S >> N;
	vector<int> v(N);

	for (i = 0; i < N; i++)
	{
		cin >> temp;
		pos = -1;
		for (j = 0; j < S; j++)
		{
			if (v[j] == temp)
			{
				pos = j;
			}
		}

		if (pos == -1)	// miss
		{
			for (j = S - 1; j >= 1; j--)
			{
				v[j] = v[j - 1];
			}
		}
		else			// hit
		{
			for (j = pos; j >= 1; j--)
			{
				v[j] = v[j - 1];
			}
		}

		v[j] = temp;	// v[0]=temp;
	}

	for (i = 0; i < S; i++)
	{
		cout << v[i] << " ";
	}

	return 0;
}
```
