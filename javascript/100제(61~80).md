const str = "aaabbbbcdddd";
let temp = {};

for (let i in str) {
  if (!temp[str[i]]) {
    temp[str[i]] = 1;
  } else {
    temp[str[i]]++;
  }
}

const obj = Object.entries(temp);
let result = "";
for (let i = 0; i < obj.length; i++) {
  result = result + obj[i][0] + obj[i][1];
}

console.log(result);
