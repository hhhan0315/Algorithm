### 21. 카드게임
```c
#include <iostream>

using namespace std;

int main()
{
	int A[10];
	int B[10];

	for (int i = 0; i < 10; i++)
	{
		cin >> A[i];
	}

	for (int i = 0; i < 10; i++)
	{
		cin >> B[i];
	}

	int count_a = 0, count_b = 0;
	int last_win = 0;

	for (int i = 0; i < 10; i++)
	{
		if (A[i] > B[i])
		{
			count_a+=3;
			last_win = 1;
		}
		else if (A[i] < B[i])
		{
			count_b+=3;
			last_win = 2;
		}
		else
		{
			count_a += 1;
			count_b += 1;
		}
	}

	cout << count_a << " " << count_b << endl;

	if (count_a == count_b)
	{
		if (last_win == 1)
			cout << "A" << endl;
		else if (last_win == 2)
			cout << "B" << endl;
		else
			cout << "D" << endl;
	}
	else if (count_a > count_b)
	{
		cout << "A" << endl;
	}
	else
	{
		cout << "B" << endl;
	}

	return 0;
}
```

### 22. 온도의 최댓값
```c
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int N, K, sum = 0, max;
	cin >> N >> K;
	vector<int> arr(N);         // 동적 할당

	for (int i = 0; i < N; i++)
	{
		cin >> arr[i];
	}

	for (int i = 0; i < K; i++)
	{
		sum += arr[i];      // 첫번째 구간 합
	}
	max = sum;

	for (int i = K; i < N; i++)
	{
		sum -= arr[i - K];  // 생각하지 못했다.
		sum += arr[i];
		if (sum > max)
			max = sum;
	}
	cout << max << endl;

	//*틀린이유* : O(n^2)은 시간 초과, O(n)으로 해결.
	return 0;
}
```
