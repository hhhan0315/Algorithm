## 14.2 구조체의 기본적인 사용법
```java
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX 41

struct person {
	char name[MAX];
	int age;
	float height;
};

int main(void) {
	int flag;

	struct person genie;	// 변수 선언하는 순간 메모리 할당

	// dot(.) is structure membership operator.
	strcpy(genie.name, "Will Smith");
	genie.age = 100;
	flag = scanf("%f", &genie.height);
	printf("%f\n", genie.height);

	// Initialization
	struct person princess = { "Naomi scott", 18, 160.0f };

	// Designated Initializers
	struct person beauty = {
		.age = 19,
		.name = "Bell",
		.height = 150.0f
	};

	// Pointer
	struct person* someone;
	someone = &genie;
	someone->age = 1001;
	printf("%s %d\n", someone->name, (*someone).age);

	struct {
		char farm[MAX];
		float price;
	} apple, apple2;

	// typedef
	typedef struct person my_person;
	my_person p1;
	typedef struct person person;
	// struct person p1;
	// person p1; 같은 의미

	return 0;
}
```
## 14.3 구조체의 메모리 할당
```c
#include <stdio.h>
#include <stdlib.h>

int main(void) {

	// 이거 자체로는 메모리 할당이 아니다.
	// Template 의미 (c++에서는 Template은 또 다른 의미)
	struct Aligned {
		int a;
		float b;
		double c;
	};

	struct Aligned a;	// 메모리 할당

	// padding  (구조체를 일정한 크기로 정렬하한다.)
	// 이때 멤버 중에서 큰 자료형 크기의 배수로 정렬하며 남는 공간을 채우는 것 : 패딩
	// 1 word : 4 bytes in x86, 8 bytes in x64
	struct Padded {
		char a;
		float b;
		double c;
		// 1 + 4 + 8 = 13 (계산하면 13이지만)
		// 4 + 4 + 8 = 16 (16으로 나옴)
	};

	struct Padded2 {
		float b;
		double c;
		char a;
		// 8 + 8 + 8 = 24 (padding이 연속으로 들어감)
	};

	return 0;
}
```
