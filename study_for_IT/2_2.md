### 44. 마구간 정하기 (이분 검색 응용)
```c
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int N;

int count(int s, vector<int> v)
{
	int cnt = 1, temp;
	int start = v[0];
	for (int i = 1; i < N; i++)
	{
		temp = v[i] - start;
		if (temp >= s)
		{
			start = v[i];
			cnt++;
		}
	}
	return cnt;
}

int main()
{
	int C, temp;
	cin >> N >> C;
	vector<int> v;

	for (int i = 0; i < N; i++)
	{
		cin >> temp;
		v.push_back(temp);
	}
	sort(v.begin(), v.end());

	int low = v[0];
	int high = v[N - 1];
	int res;

	while (low <= high)
	{
		int mid = (low + high) / 2;
		if (count(mid, v) >= C)
		{
			res = mid;
			low = mid + 1;	// 이 조건이 틀림.
		}
		else
			high = mid - 1;
	}
	cout << res << endl;

	return 0;
}
```

### 45. 공주 구하기 (조세퍼스)
```c
#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int N, K;
	cin >> N >> K;
	vector<int> visited(N + 1);

	int check = 0, pos = 0, cnt = 0;

	while (true)
	{
		pos++;
		if (pos > N)
			pos = 1;

		if (visited[pos] == 0)
		{
			cnt++;
			if (cnt == K)
			{
				visited[pos] = 1;
				cnt = 0;
				check++;
			}
		}

		if (check == (N - 1)) break;
	}

	for (int i = 1; i <= N; i++)
	{
		if (visited[i] == 0)
		{
			cout << i << endl;
			break;
		}
	}
	return 0;
}
```

### 46. 멀티 태스킹
```c
#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int N, K, sum = 0;
	cin >> N;
	vector<int> v(N + 1);

	for (int i = 1; i <= N; i++)
	{
		cin >> v[i];
		sum += v[i];
	}
	cin >> K;

	if (sum <= K)	// -1 조건도 미리 해주기.
	{
		cout << "-1" << endl;
		return 0;
	}

	int pos = 0, second = 0;
	while (true)
	{
		pos++;
		if (pos > N) pos = 1;
		if (v[pos] == 0) continue;	// (v[pos]!=0) 조건보다 훨씬 간단.
		v[pos]--;
		second++;

		if (second == K) break;
	}

	while (true)
	{
		pos++;
		if (pos > N) pos = 1;
		if (v[pos] != 0) break;
	}
	cout << pos << endl;

	return 0;
}
```

### 47. 봉우리 (2차원 배열 탐색)
```c
#include <iostream>

using namespace std;

int main()
{
	int dx[4] = { -1, 1, 0, 0 };
	int dy[4] = { 0, 0, 1, -1 };
	int arr[51][51];
	int N, flag, res = 0;
	cin >> N;


	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			cin >> arr[i][j];
		}
	}

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			flag = 0;
			for (int k = 0; k < 4; k++)
			{
				int nx = i + dx[k];
				int ny = j + dy[k];
				if (arr[i][j] <= arr[nx][ny])
				{
					flag = 1;
					break;
				}
			}

			if (flag == 0) res++;
		}
	}
	cout << res << endl;

	return 0;
}
```

### 48. 각 행의 평균과 가장 가까운 값
```c
#include <iostream>
#include <vector>
#include <limits>
#include <algorithm>
using namespace std;

int main()
{
	int arr[10][10];
	int sum, ave, temp, min, res;

	for (int i = 0; i < 9; i++)
	{
		sum = 0;
		min = INT_MAX;
		for (int j = 0; j < 9; j++)
		{
			cin >> arr[i][j];
			sum += arr[i][j];
		}
		ave = (sum / 9.0) + 0.5;	// 소수점 첫 째자리에서 반올림
		// sum/9는 소수점이 생기지않음. -> 실수 / 정수 하면 실수가 나온다.
		cout << ave << " ";

		for (int j = 0; j < 9; j++)
		{
			temp = abs(arr[i][j] - ave);	// 절대값
			if (temp < min)
			{
				min = temp;
				res = arr[i][j];
			}
			else if (temp == min)
			{
				if (res < arr[i][j]) res = arr[i][j];
			}
		}
		cout << res << endl;
	}

	return 0;
}
```
