function dfs(graph, start) {
  let visited = [];
  let stack = [start];

  while (stack.length !== 0) {
    let n = stack.pop();
    if (!visited.includes(n)) {
      visited.push(n);
      let temp = graph[n].filter((x) => !visited.includes(x));
      for (let i of temp) {
        stack.push(i);
      }
    }
  }
  return visited;
}

let visited = [];

function solution(n, visited) {
  let node = n[n.length - 1];
  let length = 0;

  if (node === end) {
    return visited.length;
  }

    return visited.length;
